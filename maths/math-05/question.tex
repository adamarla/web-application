  \marginnote[7cm]{Notice the symmetry in the figure. It is a square made up of 
     4 equal triangles in each quadrant. The required area is therefore 
     $4\times$(area of one triangle)}
  \marginnote{In fact, one does not even need to use calculus to calculate the area.
     Using standard formulae for the area of a square or a triangle should

\begin{solution}
  $\vert x \vert + \vert y \vert = 1 \Rightarrow 
  \begin{cases}
    -x-y=1 & x,y \leq 0 \\ 
    x-y=1 & x \geq 0, y \leq 0 \\
    -x+y=1 & x \leq 0, y \geq 0 \\ 
    x+y=1 & x,y \geq 0
  \end{cases}
  $ 

  The required area is that of the square $ABCD$ 
  \begin{fullwidth} 
      \begin{align}
        \text{Area} &= \int_{-1}^{0}(x+1)\ud x +
                       \left\vert \int_{-1}^0 (-1-x)\ud x\right\vert + 
                       \int_0^1 (1-x) \ud x + 
                       \left\vert \int_0^1 (x-1) \ud x \right\vert \\
                    &= \left[ \dfrac{x^2}{2}+x\right]_{-1}^0 + 
                       \left\vert\left[ -x-\dfrac{x^2}{2}\right]^0_{-1}\right\vert + 
                       \left[x-\dfrac{x^2}{2}\right]_0^1 +
                       \left\vert\left[\dfrac{x^2}{2}-x\right]_0^1\right\vert  \\
                    &= \dfrac{1}{2} + \left\vert -\dfrac{1}{2}\right\vert 
                       + \dfrac{1}{2} + \left\vert-\dfrac{1}{2}\right\vert \\ 
                    &= 2
      \end{align} 
  \end{fullwidth} 
\end{solution}
